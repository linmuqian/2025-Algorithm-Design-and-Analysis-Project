def generate_test_data(sizes):
    """生成不同规模的测试数据"""
    random.seed(RANDOM_SEED)
    datasets = {}
    for size in sizes:
        # 使用固定种子生成随机数组
        random_data = [random.randint(1, size) for i in range(size)]
        datasets[size] = random_data
    return datasets

def measure_time(algorithm, data, runs=5):
    """测量算法执行时间"""
    test_time=[]
    # 默认测试5次，然后取平均值
    for i in range(runs):
        start_time = time.time()
        algorithm(data.copy())  # 避免修改原始数据
        end_time = time.time()
        test_time.append(end_time - start_time)
    return sum(test_time)/runs

def run_benchmarks(sizes, datasets, algorithm, runs=5):
    """对排序算法在不同规模数据上进行性能测试"""
    # 储存该算法下不同数据规模的测试时间
    results = []
    
    for size in sizes:
        data = datasets[size]

        # 如果使用随机化快速排序，应该先重新设置随机种子
        if algorithm.__name__=="Randomized Sort":
            random.seed(RANDOM_SEED)

        time_taken = measure_time(algorithm, data, runs)
        results.append(time_taken)
        print(rf"{mapping[algorithm]} on data of size {size}: {time_taken:.6f} seconds")
    
    # 将结果储存到results_df中，便于之后对比
    results_df[mapping[algorithm]]=results

    return results

def plot_results(sizes, times, algorithm):
    """绘制选择排序性能图"""
    plt.figure(figsize=(10, 6))
    plt.plot(sizes, times, marker='o', color='red', label='Selection Sort')    
    plt.xlabel('Input Size (n)')
    plt.ylabel('Execution Time (seconds)')
    plt.title(rf'{algorithm} Performance vs Input Size')
    plt.legend()
    plt.grid(True)
    plt.show()
